44 int bi_get_nafb_std(sbyte *naf, word_t *a) {
45  uint i = 0;
46  word_t carry, tmp;
47 
48  carry = 0;
49 
50  while (bi_compare(a, bi_zero) > 0) {
51  if (bi_is_odd(a)) {
52  // at this point, no carry can exist
53  naf[i] = 2 - (a[0]&3);
54  tmp = a[0] >> (BITS_PER_WORD-1);
55  a[0] -= (word_t) naf[i];
56  carry = tmp & ~(a[0] >> (BITS_PER_WORD-1));
57  } else {
58  naf[i] = 0;
59  }
60 
61  // carry propagation
62  tmp = a[1] & 0x01 & carry;
63  a[1] = a[1] ^ carry;
64  carry = tmp;
65 
66  bi_shift_right_one(a, a);
67  i++;
68  }
69  return i;
70 }

151  uint i;
152  word_t c, carry, r;
153 
154  carry = 0;
155  for (i = 0; i < length; i++) {
156  r = *(a+i) - *(b+i);
157  c = (*(a+i) < *(b+i));

159  carry = c | (!r && carry);

253 void bi_shift_right_var_std(word_t *res, const word_t *a, const uint right, const length_t length) {
254  int i;
255  uint wordNum = right >> LD_BITS_PER_WORD;
256  uint bitNum = right & (BITS_PER_WORD-1);
257  word_t shift_in;
258 
259  for(i = length-1; i >= (int)(length-wordNum); i--)
260  res[i] = 0;
261 
262  if (bitNum == 0) {
263  a += wordNum;
264  for (i = 0; i < (int)(length-wordNum); i++) {
265  *(res+i) = *(a+i);
266  }
267  } else {
268  a += wordNum;
269  for (i = 0; i < (int)(length-wordNum -1); i++) {
270  shift_in = *(a+i+1) << (BITS_PER_WORD - bitNum);
271  *(res+i) = (*(a+i) >> bitNum) | shift_in;
272  }
273  *(res+i) = *(a+i) >> bitNum;
274  }
275 }

400 void bi_multiply_var_std(word_t *result, const word_t *a, const word_t *b, const length_t length_a, const length_t length_b)
401 {
402  dword_t tmp;
403  word_t carry;
404  uint i, j;
405  bi_clear_var(result, length_a + length_b);
406 
407  // many carries ... bad performance -> comba is better :)
408  for (i = 0; i < length_a; i++) {
409  carry = 0;
410  for (j = 0; j < length_b; j++) {
411  tmp = *(result+i+j);
412  tmp += ((dword_t)*(a+i)) * ((dword_t)*(b+j));
413  tmp += carry;
414  *(result+i+j) = (word_t) tmp;
415  carry = (tmp >> BITS_PER_WORD);
416  }
417  *(result + i + length_b) = carry;
418  }
419 }
