64 void cprng_get_bytes(void *buffer, length_t size) {
65  byte carry, c0, c1, r0, r1;
66  int i, j;
67  byte hash[HASH_LEN];
68  byte *buf = (byte*)buffer;
69 
70  j = 0;
71  while (j < size) {
72  // x = G(t, XKEY)
73  cprng_one_way_func(hash, CPRNGState, CPRNG_SIZE);
74 
75  // XKEY = (XKEY + x + 1) mod 2^b, where b = number of bits of CPRNG
76  carry = 1;
77  for (i = 0; i < CPRNG_SIZE; i++) {
78  r0 = (byte) (CPRNGState[i] + hash[i]);
79  c0 = (byte) (r0 < hash[i]);
80  r1 = (byte) (r0 + carry);
81  c1 = (byte) (r1 < r0);
82  carry = (byte) (c0 | c1);
83  CPRNGState[i] = r1;
84  }
85  for (i = 0; i < CPRNG_SIZE && j < size; i++) {
86  buf[j] = hash[i];
87  j++;
88  }
89  }
90 }
