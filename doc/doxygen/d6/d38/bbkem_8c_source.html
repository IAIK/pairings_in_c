86 void derive_private_key(bbkem_pk *upk, bbkem_msk *msk, bbkem_public *params, const char *id)
87 {
88  bigint_t id_bi;
89  fp_t r, k;
90 
91  do {
92  cprng_get_bytes(r, BI_BYTES); fp_rdc_n(r);
93  } while (bi_compare(r, bi_zero) == 0);
94 
95  ecfp2_mul(&(upk->d1), (const ecfp2_pt)&ECFP2_GENERATOR, r);
96 
97  hash_id(id_bi, id);
98 
99  fp_mul_barett_var(k, id_bi, msk->alpha, EC_PARAM_N, EC_PARAM_MU_N);
100  fp_add_var(k, k, msk->gamma, EC_PARAM_N);
101  fp_mul_barett_var(k, k, r, EC_PARAM_N, EC_PARAM_MU_N);
102  fp_mul_barett_var(r, msk->alpha, msk->beta, EC_PARAM_N, EC_PARAM_MU_N);
103  fp_add_var(k, k, r, EC_PARAM_N);
104  ecfp2_mul(&(upk->d0), (const ecfp2_pt)&ECFP2_GENERATOR, k);
105 }

115 void encapsulate_key(byte *key, bbkem_ciphertext *cipher, bbkem_public *params, const char *id) {
116  bigint_t s, factor, id_bi;
117  ecpoint_fp p;
118  fp12_t k;
119 
120  cprng_get_bytes(s, BI_BYTES);
121 
122  fp_rdc_n(s); // reduce random number s by group order (constant time!)
123 
124  hash_id(id_bi, id);
125 
126  // this multiplication needs to have constant runtime
127  fp_mul_barett_var(factor, id_bi, s, EC_PARAM_N, EC_PARAM_MU_N);
128 
129  // s*g3 + factor*g1 ... => interleaved calculation?
130  ecfp_mul(&(cipher->c1), (const ecfp_pt) &(params->g3), s);
131 
132  ecfp_mul(&(cipher->c0), (const ecfp_pt) &(params->g1), factor);
133  ecfp_add_affine(&(cipher->c1), &(cipher->c1), &(cipher->c0));
134 
135  ecfp_mul(&(cipher->c0), (const ecfp_pt) &ECFP_GENERATOR, s);
136 
137  fp12_exp_cyclotomic(k, (const fp4_t*)params->v0, s);
138 #ifdef MONTGOMERY_ARITHMETIC
139  ecfp_from_montgomery(&(cipher->c0), &(cipher->c0));
140  ecfp_from_montgomery(&(cipher->c1), &(cipher->c1));
141  fp12_from_montgomery(k, (const fp4_t *) k);
142 #endif
143 
144  hash_key(key, k);
145 }

157 void decapsulate_key(byte *key, bbkem_ciphertext *cipher, bbkem_public *params, bbkem_pk *upk) {
158  fp12_t p_res;
159 
160 #ifdef MONTGOMERY_ARITHMETIC
161  ecpoint_fp c0, c1;
162 
163  ecfp_to_montgomery(&c0, &(cipher->c0));
164  ecfp_to_montgomery(&c1, &(cipher->c1));
165  ecfp_neg_affine(&c1);
166  pbc_map_opt_ate_mul(p_res, &c0, &(upk->d0), &c1, &(upk->d1));
167  fp12_from_montgomery(p_res, (const fp4_t*) p_res);
168 #else
169  pbc_map_opt_ate_div(p_res, &(cipher->c0), &(upk->d0), &(cipher->c1), &(upk->d1));
170 #endif
171 
172  hash_key(key, p_res);
173 }
