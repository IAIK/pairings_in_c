75 void benchmark_compute(int iterations) {

77  bm_result = (bm_stop - bm_start - BENCH_OVERHEAD) / iterations;
78 #elif (BENCHMARK_SRC == BM_CLOCK)
79  bm_result = ((long)bm_stop.tv_sec - (long)bm_start.tv_sec) * 1000000000;
80  bm_result += (bm_stop.tv_nsec - bm_start.tv_nsec);
