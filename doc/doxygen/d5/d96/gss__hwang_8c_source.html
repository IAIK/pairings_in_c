287 void hwang_generate_usk(hwang_signing_key_ptr usk, hwang_public_parameters_ptr data)
288 {
289  bigint_t bi_tmp, bi_tmp1, inv, bi_tmp3;
290  word_t bi_tmp_mul[2*BI_WORDS];
291  ecpoint_fp G1_tmp;
292 
293  // x, y, z \in_R \mathbb{Z}_p
294  do {
295  cprng_get_bytes(usk->x, BI_BYTES); fp_rdc_n(usk->x);
296  } while (bi_compare(usk->x, bi_zero) == 0);
297 
298  do {
299  cprng_get_bytes(usk->y, BI_BYTES); fp_rdc_n(usk->y);
300  } while (bi_compare(usk->y, bi_zero) == 0);
301 
302  do {
303  cprng_get_bytes(usk->z, BI_BYTES); fp_rdc_n(usk->z);
304  } while (bi_compare(usk->z, bi_zero) == 0);
305 
306 
307  // A = g1 * g2^{-y} * w^{-z}
308  bi_subtract(bi_tmp, data->order_p, usk->y); // negate y
309  bi_subtract(bi_tmp1, data->order_p, usk->z); // negate z
310  ecfp_mul(&usk->A, &data->g2, bi_tmp); // A = g2^{-y}
311  ecfp_mul(&G1_tmp, &data->w, bi_tmp1); // = w^{-z}
312  ecfp_add_affine(&usk->A, &usk->A, &G1_tmp);
313  ecfp_add_affine(&usk->A, &usk->A, &data->g1);
314 
315 
316  bi_add(bi_tmp, data->theta, usk->x); fp_rdc_n(bi_tmp);
317 
318  fp_inv_n(inv, bi_tmp);
319 
320  ecfp_mul(&usk->A, &usk->A, inv); // A = (g1 * g2^{-y} * w^{-z})^{1/(theta + x)}
321 
322 #if DEBUG_PRINT == 1
323  print("\n\n// -------------- USK --------------\n");
324 
325  PRINT_BIGINT("USK_x: ", usk->x);
326  PRINT_BIGINT("USK_y: ", usk->y);
327  PRINT_BIGINT("USK_z: ", usk->z);
328  PRINT_G1("USK_A: ", usk->A);
329 #endif
330 }

341 sbyte hwang_verify(hwang_public_parameters_ptr data, hwang_signature_ptr sig)
342 {
343  hashState state;
344  unsigned char block[64];
345  unsigned char hash[HASH_BYTES];
346  unsigned char msg[1] = {0x01};
347 
348  sbyte ret = -1;
349  ecpoint_fp R_1, R_3, g1_tmp1, g1_tmp2;
350 
351  fp12_t R_2, gt_tmp1, gt_tmp2, gt_tmp3, gt_tmp4;
352 
353  bigint_t bi_tmp, c;
354  word_t bigint_tmp_dbl[2*BI_WORDS];
355 
356  // BEGIN: Check signature
357 
358  ecfp_mul(&R_1, &data->u, sig->s_alpha); // R_1 = u^{s_{alpha}}
359 
360  bi_subtract(bi_tmp, data->order_p, sig->c); // negate c (mod order p);
361 
362  ecfp_mul(&g1_tmp1, &sig->D_1, bi_tmp);
363  ecfp_add_affine(&R_1, &R_1, &g1_tmp1); // R_1 = u^{s_{alpha}} * D_1^{-c}
364 
365 #if DEBUG_PRINT == 1
366  PRINT_G1("[verify] R_1: ", R_1);
367 #endif
368 
369 
370  ecfp_mul(&g1_tmp1, &sig->D_2, sig->s_x);
371  bi_subtract(bi_tmp, data->order_p, sig->s_alpha); // negate s_alpha (mod order p)
372  ecfp_mul(&g1_tmp2, &data->w, bi_tmp);
373  pbc_map_opt_ate_mul(gt_tmp1, &g1_tmp1, &data->h1, &g1_tmp2, & data->h_theta); // R_2 = e(D_2, h_1)^{s_x} * e(w, h_theta)^{-s_{alpha}}
374 
375  bi_subtract(bi_tmp, data->order_p, sig->s_gamma); // negate s_gamma (mod order p)
376  ecfp_mul(&g1_tmp1, &data->w, bi_tmp);
377  ecfp_mul(&g1_tmp2, &data->g2, sig->s_y);
378  pbc_map_opt_ate_mul(gt_tmp2, &g1_tmp1, &data->h1, &g1_tmp2, &data->h1);
379  fp12_mul_std(gt_tmp3, (const fp4_t*)gt_tmp1, (const fp4_t*)gt_tmp2);
380 
381  pbc_map_opt_ate_div(gt_tmp1, &sig->D_2, &data->h_theta, &data->g1, &data->h1);
382  fp12_exp_std(gt_tmp2, (const fp4_t*)gt_tmp1, sig->c);
383  fp12_mul_std(R_2, (const fp4_t*)gt_tmp3, (const fp4_t*)gt_tmp2);
384 
385 #if DEBUG_PRINT == 1
386  PRINT_GT("[verify] R_2:", R_2);
387 #endif
388 
389  ecfp_mul(&R_3, &data->g, sig->s_y);
390  ecfp_mul(&g1_tmp1, &data->d, sig->s_alpha);
391  ecfp_add_affine(&R_3, &R_3, &g1_tmp1);
392  bi_subtract(bi_tmp, data->order_p, sig->c);
393  ecfp_mul(&g1_tmp1, &sig->D_3, bi_tmp);
394  ecfp_add_affine(&R_3, &R_3, &g1_tmp1);
395 
396 
397 #if DEBUG_PRINT == 1
398  PRINT_G1("[verify] R_3:", R_3);
399 #endif
400 
401  hash_init(&state);
402 
403  // hash message M
404  memset(block, 0, 64); memcpy(block, msg, 1); hash_update(&state, block, 64);
405 
406  // hash D1
407  hash_update_G1(&state, &sig->D_1);
408 
409  // hash D2
410  hash_update_G1(&state, &sig->D_2);
411 
412  // hash D3
413  hash_update_G1(&state, &sig->D_3);
414 
415  // hash R1
416  hash_update_G1(&state, &R_1);
417 
418  // hash R2
419  hash_update_GT(&state, R_2);
420 
421  // hash R3
422  hash_update_G1(&state, &R_3);
423 
424  hash_final(&state , hash, HASH_BYTES);
425 
426  bi_clear_var_std(bigint_tmp_dbl, 2*BI_WORDS); bi_copy_var_std(bigint_tmp_dbl, (const word_t *) hash, HASH_BYTES/4);
427  fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
428  bi_copy_var_std(c, bigint_tmp_dbl, BI_WORDS);
429 
430 #if DEBUG_PRINT == 1
431  PRINT_BIGINT("[verify]: c ", c);
432  PRINT_BIGINT("[signature: c ", sig->c);
433 #endif
434 
435  if (bi_compare(sig->c, c) == 0)
436  ret = 1;
437 
438  return ret;
439 }
