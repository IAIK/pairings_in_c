70 void hash_update_message(hashState *state, const char *message)
71 {
72  byte block[BLOCK_BYTES];
73  unsigned int message_bytes = strlen(message);
74  if(message_bytes < BLOCK_BYTES) {
75  memset(block, 0, BLOCK_BYTES);
76  memcpy(block, message, message_bytes);
77  hash_update(state, block, BLOCK_BITS);
78  } else {
79  unsigned int i = 0;
80  for(i = 0; message_bytes > BLOCK_BYTES; i++) {
81  memset(block, 0, BLOCK_BYTES);
82  memcpy(block, (message + i*BLOCK_BYTES), BLOCK_BYTES);
83  hash_update(state, block, BLOCK_BITS);
84  message_bytes -= BLOCK_BYTES;
85  }
86  memset(block, 0, BLOCK_BYTES);
87  memcpy(block, (message + i*BLOCK_BYTES), message_bytes);
88  hash_update(state, block, BLOCK_BITS);
89  }
90 
91 }
