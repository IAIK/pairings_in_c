118 void ecfp2_add_proj_std(ecfp2_proj_pt res, const ecfp2_proj_pt a, const ecfp2_pt b) {
119  // calculation based on jacobian coordinates
120  // b in projective coordinates is (b.x, b.y, 1)
121 
122  if (a->infinity && !b->infinity) { // P + Infinity = P
123  fp2_copy(res->x, (const fp_t*) b->x);
124  fp2_copy(res->y, (const fp_t*) b->y);
125  fp_copy(res->z[0], FP_ONE);
126  fp_copy(res->z[1], bi_zero);
127  res->infinity = 0;
128  return;
129  } else if (!a->infinity && b->infinity) {
130  fp2_copy(res->x, (const fp_t*) a->x);
131  fp2_copy(res->y, (const fp_t*) a->y);
132  fp_copy(res->z[0], FP_ONE);
133  fp_copy(res->z[1], bi_zero);
134  res->infinity = 0;
135  return;
136  } else if (a->infinity && b->infinity) {
137  res->infinity = 1;
138  return;
139  }
140 
141  fp2_t t0, t1, t2, t3;
142 
143  fp2_sqr(t0, (const fp_t*) a->z); // t0 = z²
144  fp2_mul(t1, (const fp_t*) t0, (const fp_t*) a->z); // t1 = z³
145 
146  fp2_mul(t2, (const fp_t*) t0, (const fp_t*) b->x); // t2 = b->x*z²
147  fp2_sub(t0, (const fp_t*) t2, (const fp_t*) a->x); // t0 = b->x*z² - a->x
148 
149  fp2_mul(t1, (const fp_t*) t1, (const fp_t*) b->y);
150  fp2_sub(t1, (const fp_t*) t1, (const fp_t*) a->y); // t1 = b->y*z³ - a->y
151 
152  if (bi_compare(t0[0], bi_zero) == 0 && bi_compare(t0[1], bi_zero) == 0) {
153  if (bi_compare(t1[1], bi_zero) == 0 && bi_compare(t1[1], bi_zero) == 0) {
154  // point a is equal to point b (doubling!)
155  ecfp2_dbl_proj(res, a);
156  } else { // otherwise its point (1,1,0) -> infinity
157  res->infinity = 1;
158  }
159  return;
160  } else {
161  res->infinity = 0;
162  }
163 
164  fp2_mul(res->z, (const fp_t*) t0, (const fp_t*) a->z); // res->z = (b->x*z² - a->x) * a->z
165 
166  fp2_add(t2, (const fp_t*) t2, (const fp_t*) a->x); // t2 = b->x * z² + a->x
167  fp2_sqr(t3, (const fp_t*) t0); // t3 = (b->x*z² - a->x)²
168  fp2_mul(t2, (const fp_t*) t2, (const fp_t*) t3); // t2 = (b->x*z² - a->x)² * (b->x * z² + a->x)
169  fp2_mul(t0, (const fp_t*) t0, (const fp_t*) t3); // t0 = (b->x*z² - a->x)³
170  fp2_mul(t0, (const fp_t*) t0, (const fp_t*) a->y); // t0 = a->y * (b->x*z² - a->x)³
171  fp2_mul(t3, (const fp_t*) t3, (const fp_t*) a->x); // t3 = a->x * (b->x*z² - a->x)²
172 
173  fp2_sqr(res->x, (const fp_t*) t1); // res->x = (b->y*z³ - a->y)²
174  fp2_sub(res->x, (const fp_t*) res->x, (const fp_t*) t2);// res->x = (b->y*z³ - a->y)² -
175  // (b->x*z² - a->x)² * (b->x * z² + a->x)
176  fp2_sub(t3, (const fp_t*) t3, (const fp_t*) res->x); // a->x * (b->x*z² - a->x)² - res->x
177  fp2_mul(t3, (const fp_t*) t3, (const fp_t*) t1);
178  fp2_sub(res->y, (const fp_t*) t3, (const fp_t*) t0); // (b->y*z³ - a->y)(a->x * (b->x*z² - a->x)² - res->x) -
179  // a->y * (b->x*z² - a->x)³
180 }

247 void ecfp2_dbl_proj_std(ecfp2_proj_pt res, const ecfp2_proj_pt a) {
248  // calculation based on jacobian coordinates
249 
250  if (a->infinity) {
251  res->infinity = 1;
252  return;
253  } else {
254  res->infinity = 0;
255  }
256 
257  fp2_t t0, t1, t2;
258 
259  fp2_sqr(t0, (const fp_t*) a->x);
260 
261 #ifdef OPTIMIZE_ZERO_A
262  fp2_dbl(t1, (const fp_t*) t0);
263  fp2_add(t1, (const fp_t*) t1, (const fp_t*) t0); // t1 = a*z⁴ + 3x²
264 #else
265  fp2_sqr(t1, a->z);
266  fp2_sqr(t1, (const fp_t*) t1); // t1 = z⁴
267 
268  fp_clear(t2[1]); // calc a' of twisted curve
269  fp_copy(t2[0], EC_PARAM_A);
270  fp2_div_qnr(t2, (const fp_t*) t2);
271  fp2_mul(t1, t1, t2); // t1 = a*z⁴
272 
273  fp2_add(t1, t1, t0);
274  fp2_add(t1, t1, t0);
275  fp2_add(t1, t1, t0); // t1 = a*z⁴ + 3x²
276 #endif
277 
278  fp2_dbl(t0, (const fp_t*) a->y); // t0 = 2 * y
279  fp2_mul(res->z, (const fp_t*) t0, (const fp_t*) a->z); // res->z = 2*y*z
280 
281  fp2_sqr(t0, (const fp_t*) t0); // t0 = 4 * y²
282  fp2_mul(t2, (const fp_t*) t0, (const fp_t*) a->x); // t2 = 4*x*y²
283 
284  fp2_copy(res->y, (const fp_t*) t1); // res->y = (3x² + az⁴)
285  fp2_dbl(t1, (const fp_t*) t2); // t1 = 8*x*y²
286 
287  fp2_sqr(res->x, (const fp_t*) res->y); // res->x = (3x² + az⁴)²
288  fp2_sub(res->x, (const fp_t*) res->x, (const fp_t*) t1);// res->x = (3x² + az⁴)² - 8*x*y²
289 
290  fp2_sub(t2, (const fp_t*) t2, (const fp_t*) res->x); // 4xy² - res->x
291  fp2_mul(res->y, (const fp_t*) res->y, (const fp_t*) t2);// res->y = (3x² + az⁴)(4xy² - res->x)
292 
293  fp2_sqr(t0, (const fp_t*) t0);
294  fp2_hlv(t0, (const fp_t*) t0); // t0 = 8 * y⁴
295 
296  fp2_sub(res->y, (const fp_t*) res->y, (const fp_t*) t0);// res->y = (3x² + az⁴)(4xy² - res->x) - 8y⁴
297 }

561 void ecfp2_mul_montyladder_std(ecfp2_pt res, const ecfp2_pt a, const fp_t k) {
562  int msb = bi_get_msb(k);
563  int i;
564  fp2_t x1, x2, z;
565  fp_t *x[2];
566  word_t bit;
567 
568  x[0] = x1;
569  x[1] = x2;
570 
571  if (ecfp2_verify_coordinates(a->x, a->y))
572  return;
573 
574  cprng_get_bytes(z, 2*FP_BYTES);
575  fp_rdc(z[0]);
576  fp_rdc(z[1]);
577 
578  fp2_mul(x[1], (const fp_t*) a->x, (const fp_t*)z);
579  fp2_mul(x[0], (const fp_t*) a->y, (const fp_t*)z);
580 
581  if (ecfp2_verify_homogeneous_projective_coordinates(x[1], x[0], z))
582  return;
583 
584  fp2_clear(x[0]);
585 
586  ecfp2_add_dbl_coz_std(x1, x2, z, (const fp_t*) a->x);
587 
588  fp2_mul(x1, (const fp_t*) a->x, (const fp_t*) z);
589 
590  for (i = msb - 1; i >= 0; i--) {
591  bit = bi_test_bit(k, i);
592  ecfp2_add_dbl_coz_std(x[~bit & 1], x[bit], z, (const fp_t*) a->x);
593  }
594 
595  ecfp2_recover_full_coord_coz_std(x[0], x[1], z, (const fp_t*) a->x, (const fp_t*) a->y);
596 
597  if (ecfp2_verify_homogeneous_projective_coordinates(x[0], x[1], z))
598  return;
599 
600  fp2_inv(z, (const fp_t*) z);
601  fp2_mul(res->x, (const fp_t*) x[0], (const fp_t*) z);
602  fp2_mul(res->y, (const fp_t*) x[1], (const fp_t*) z);
603  res->infinity = 0;
604 
605  if (ecfp2_verify_coordinates(res->x, res->y))
606  return;
607 }

753 void ecfp2_frobenius_map_std(ecfp2_pt res, const ecfp2_pt a, const word_t i) {
754  fp2_t q, q2, q3;
755  bigint_t e;
756  uint j;
757 
758  fp_copy(q[0], FP_ONE);
759  fp_clear(q[1]);
760  fp2_mul_qnr(q2, (const fp_t*) q);
761  bi_copy(e, PRIME_P);
762  bi_subtract(e, e, bi_one);
763  bi_div3(e,e);
764  bi_shift_right_one(e,e);
765  fp2_exp(q, (const fp_t*) q2, e); // qnr^{(p-1)/6}
766

768  fp2_mul(q3, (const fp_t*) q, (const fp_t*) q2);
769 
770  ecfp2_copy(res, a); // TODO: extract i = 1 to save copy
771 
772  // map coefficients and multiply with the corresponding roots resulting from the
773  // frobenius map in GFp12, i.e. untwist the curve, perform frobenius map, and twist again
774  for (j = 0; j < i; j++) {
775  fp2_frobenius_map(res->x, (const fp_t*) res->x, 1);
776  fp2_mul(res->x, (const fp_t*) res->x, (const fp_t*) q2);

806 void ecfp2_hash_to_point_std(ecfp2_pt res, const bigint_t t) {
807  fp_t sqrtM3;
808  fp2_t w, t0;
809  fp2_t x[3];
810  ecpoint_fp2 px;
811  ecpoint_fp2_proj tp;
812 
813  // Firstly: Hash to a random point
814 
815  fp_dbl(sqrtM3, FP_ONE);
816  fp_add(sqrtM3, sqrtM3, FP_ONE);
817  fp_neg(sqrtM3, sqrtM3);
818  fp_sqrt(sqrtM3, sqrtM3); // sqrt(-3)
819 
820  fp_clear(w[1]);
821  fp_sqr(w[0], t);
822  fp_add(w[0], w[0], FP_ONE);
823 
824  fp_clear(t0[1]); // calc b' of twisted curve
825  fp_copy(t0[0], EC_PARAM_B);
826  fp2_div_qnr(t0, (const fp_t*) t0);
827 
828  fp2_add(w, (const fp_t*) w, (const fp_t*) t0);
829  fp2_inv(w, (const fp_t*) w);
830  fp2_mulfp(w, (const fp_t*) w, sqrtM3);
831  fp2_mulfp(w, (const fp_t*) w, (const word_t*) t);
832 
833  fp2_sqr(x[2], (const fp_t*) w);
834  fp2_inv(x[2], (const fp_t*) x[2]);
835  fp_add(x[2][0], x[2][0], FP_ONE);
836 
837  fp_clear(x[0][1]);
838  fp_sub(x[0][0], sqrtM3, FP_ONE);
839  fp2_hlv(x[0], (const fp_t*) x[0]);
840  fp2_mulfp(w, (const fp_t*) w, (const word_t*) t);
841  fp2_sub(x[0], (const fp_t*) x[0], (const fp_t*) w);
842 
843  fp_clear(x[1][1]);
844  fp_neg(x[1][0], FP_ONE);
845  fp2_sub(x[1], (const fp_t*) x[1], (const fp_t*) x[0]);
846 
847  // implementation without blinding (as suggested by Fouque, Tibouchi)
848  fp2_sqr(w, (const fp_t*) x[0]);
849  fp2_mul(w, (const fp_t*) w, (const fp_t*) x[0]);
850  fp2_add(w, (const fp_t*) w, (const fp_t*) t0); // t0 = b'
851 
852  int alpha = fp2_legendre((const fp_t*) w);
853 
854  fp2_sqr(w, (const fp_t*) x[1]);
855  fp2_mul(w, (const fp_t*) w, (const fp_t*) x[1]);
856  fp2_add(w, (const fp_t*) w, (const fp_t*) t0);
857 
858  int beta = fp2_legendre((const fp_t*) w);
859 
860  int idx = ((alpha-1)*beta % 3);
861  if (idx < 0)
862  idx+=3;
863 
864  res->infinity = 0;
865  fp2_copy(res->x, (const fp_t*) x[idx]);
866  fp2_sqr(w, (const fp_t*) x[idx]);
867  fp2_mul(w, (const fp_t*) w, (const fp_t*) x[idx]);
868  fp2_add(w, (const fp_t*) w, (const fp_t*) t0);
869  fp2_sqrt(res->y, (const fp_t*) w);
870 
871  if (fp_legendre(t) == -1)
872  fp2_neg(res->y, (const fp_t*) res->y);
873 
874  // Secondly: multiply by cofactor to have correct order
875 
876  ecfp2_mul(&px, res, EC_PARAM_X);
877 #ifdef NEGATIVE_PARAM_X
878  ecfp2_neg_affine(&px);
879 #endif
880  ecfp2_frobenius_map(res, res, 3);
881  ecfp2_get_jacobian_projective(&tp, res);
882  ecfp2_add_proj(&tp, &tp, &px);
883 
884  ecfp2_frobenius_map(res, &px, 2);
885  ecfp2_add_proj(&tp, &tp, res);
886 
887  ecfp2_dbl_affine(res, &px);
888  ecfp2_add_affine(res, res, &px);
889  ecfp2_frobenius_map(res, res, 1);
890  ecfp2_add_proj(&tp, &tp, res);
891 
892  ecfp2_get_affine_from_jacobian(res, &tp);
893 }
