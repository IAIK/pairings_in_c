294 sbyte sgs_verify(group_public_key gpk, sdh_signiture sig, const char *message) {
295  bigint_t bi_tmp, bi_tmp1, c;
296  ecpoint_fp G1_tmp1, G1_tmp2, R_1, R_2, R_4, R_5;
297  fp12_t GT_tmp1, GT_tmp2, GT_tmp3, R_3;
298  unsigned char hash[HASH_BYTES];
299  word_t bigint_tmp_dbl[2*BI_WORDS], bigint_tmp_dbl2[2*BI_WORDS];
300 
301  hashState state;
302 
303  // R1
304  bi_subtract(bi_tmp, EC_PARAM_N, sig.c);
305  ecfp_mul(&G1_tmp1, &sig.T_1, bi_tmp);
306  ecfp_mul(&G1_tmp2, &gpk.u, sig.s_alpha);
307  ecfp_add_affine(&R_1, &G1_tmp1, &G1_tmp2);
308 
309  // R2
310  ecfp_mul(&G1_tmp1, &sig.T_2, bi_tmp);
311  ecfp_mul(&G1_tmp2, &gpk.v, sig.s_beta);
312  ecfp_add_affine(&R_2, &G1_tmp1, &G1_tmp2);
313 
314  // R3
315  ecfp_mul(&G1_tmp1, &sig.T_3, sig.s_x);
316 
317  bi_subtract(bi_tmp, EC_PARAM_N, sig.s_delta1);
318  bi_subtract(bi_tmp1, EC_PARAM_N, sig.s_delta2);
319  bi_add(bi_tmp, bi_tmp, bi_tmp1); fp_rdc_n(bi_tmp);
320  ecfp_mul(&G1_tmp2, &gpk.h, bi_tmp);

328  pbc_map_opt_ate(GT_tmp2, &G1_tmp2, &gpk.w);

332  ecfp_mul(&G1_tmp2, &gpk.g1, sig.c);
333

335

338 
339  fp12_mul_std(GT_tmp2, (const fp4_t*)GT_tmp1, (const fp4_t*)GT_tmp2);
340 
341  fp12_mul_std(R_3, (const fp4_t*)GT_tmp3, (const fp4_t*)GT_tmp2);
342 
343  // R4
344  bi_subtract(bi_tmp, EC_PARAM_N, sig.s_delta1);
345  ecfp_mul(&G1_tmp1, &gpk.u, bi_tmp);
346  ecfp_mul(&G1_tmp2, &sig.T_1, sig.s_x);
347  ecfp_add_affine(&R_4, &G1_tmp1, &G1_tmp2);
348

353  ecfp_add_affine(&R_5, &G1_tmp1, &G1_tmp2);

362

364  hash_update_G1(&state, &sig.T_2);
365

387  bi_copy_var_std(c, bigint_tmp_dbl, BI_WORDS);
