196 void fp_rdc_monty_lazy(fp_t res, word_t *t, const bigint_t mod, const word_t n0, const bigint_t multipleMod) {
197  int i, j, neg;
198  word_t m, c, ct, cp;
199  dword_t r;
200  ct = 0;
201 
202  // Taken from the separated operand scanning method
203  for (i = 0; i < BI_WORDS; i++) {
204  c = 0;
205  m = t[i] * n0;
206  for (j = 0; j < BI_WORDS; j++) {
207  r = t[i+j];
208  r += ((dword_t)m*(dword_t)mod[j]);
209  r += c;
210  t[i+j] = (r & WORD_MAX);
211  c = (r >> BITS_PER_WORD);
212  }
213  for (j = i + BI_WORDS; j < 2*BI_WORDS; j++) {
214  r = t[j];
215  r += c;
216  t[j] = (r & WORD_MAX);
217  c = (r >> BITS_PER_WORD);
218  }
219  ct += c;
220  }
221 
222  if (ct > 0) {
223  bi_subtract(res, t+BI_WORDS, multipleMod);
224  } else {
225  bi_copy(res, t+BI_WORDS);
226  }

236 void fp_rdc_monty_std(fp_t res, word_t *t, const bigint_t mod, const word_t n0)
237 {
238  int i, j, neg;
239  word_t m, c, ct, cp;
240  dword_t r;
241  ct = 0;
242 
243  // Taken from the separated operand scanning method
244  for (i = 0; i < BI_WORDS; i++) {
245  c = 0;
246  m = t[i] * n0;
247  for (j = 0; j < BI_WORDS; j++) {
248  r = t[i+j];
249  r += ((dword_t)m*(dword_t)mod[j]);
250  r += c;
251  t[i+j] = (r & WORD_MAX);
252  c = (r >> BITS_PER_WORD);
253  }
254  for (j = i + BI_WORDS; j < 2*BI_WORDS; j++) {
255  r = t[j];
256  r += c;
257  t[j] = (r & WORD_MAX);
258  c = (r >> BITS_PER_WORD);
259  }
260  ct += c;
261  }
262 
263 // LAZY reduction in Fp2 multiplication has impact on reduction step
264 #ifdef LAZY_REDUCTION
265  c = bi_subtract(t, t+BI_WORDS, mod);
266  cp = bi_add(res, t+BI_WORDS, mod);
267 
268  if (!c) {
269  if (cp)
270  bi_copy(res, res);
271  else
272  bi_copy(res, t);
273  } else {
274  bi_copy(res, t+BI_WORDS);
275  }
276 #else
277  c = bi_subtract(t, t+BI_WORDS, mod);
278  if (ct || !c) {
279  bi_copy(res, t);
280  } else {
281  bi_copy(res, t+BI_WORDS);
282  }
283 #endif

326 void fp_mul_monty_fips_std(fp_t res, const fp_t a, const fp_t b, const bigint_t n, const word_t n0) {
327  int i, j;
328  word_t t[3];
329  word_t tmp[3];
330  fp_t m;
331 
332  bi_clear_var(t, 3);
333  bi_clear_var(tmp, 3);
334 
335  for (i = 0; i < BI_WORDS; i++) {
336  for (j = 0; j < i; j++) {
337  bi_multiply_var(tmp, a+j, b+i-j, 1, 1);
338  bi_add_var(t, tmp, t, 3);
339  bi_multiply_var(tmp, m+j, n+i-j, 1, 1);
340  bi_add_var(t, tmp, t, 3);
341  }
342  bi_multiply_var(tmp, a+i, b, 1, 1);
343  bi_add_var(t, tmp, t, 3);
344  *(m+i) = t[0]*n0; // mod W
345  bi_multiply_var(tmp, m+i, n, 1, 1);
346  bi_add_var(t, tmp, t, 3);
347  t[0] = t[1];
348  t[1] = t[2];
349  t[2] = 0;
350  }
351  for (i = BI_WORDS; i < (2*BI_WORDS); i++) {
352  for (j = i-BI_WORDS+1; j < BI_WORDS; j++) {
353  bi_multiply_var(tmp, a+j, b+i-j, 1, 1);
354  bi_add_var(t, tmp, t, 3);
355  bi_multiply_var(tmp, m+j, n+i-j, 1, 1);
356  bi_add_var(t, tmp, t, 3);
357  }
358  m[i-BI_WORDS] = t[0];
359  t[0] = t[1];
360  t[1] = t[2];
361  t[2] = 0;
362  }
363 
364  bi_copy(res, m);
365 
366  if (bi_compare(res, n) >= 0 || t[0])
367  bi_subtract(res, res, n);
368 }

378 void fp_mul_monty_std(fp_t res, const fp_t a, const fp_t b, const bigint_t modulo, const bigint_t n0) {
379 
380  // Montgomery multiplication
381  word_t tmp[2 * FP_WORDS];
382  word_t tmp2[3 * FP_WORDS];
383  word_t tmp3[2 * FP_WORDS];
384 
385  // Multiplication
386  bi_multiply(tmp, a, b); // 2*FP_WORDS
387 
388  // Reduction
389  // t * n0
390  bi_multiply_var(tmp2, tmp, n0, 2 * FP_WORDS, FP_WORDS); // 3*FP_WORDS
391 
392  // Multiply by n(=modulo)
393  bi_multiply(tmp3, tmp2, modulo); // 2*FP_WORDS
394 
395  // Add t
396  int carry = bi_add_var(tmp, tmp, tmp3, 2 * FP_WORDS); // 2*FP_WORDS
397 
398  // Divide by R(right-shift by R bit)
399  bi_shift_right_var(tmp3, tmp, FP_BITS,
400  2 * FP_WORDS); // FP_WORDS
401 
402  if (carry) {
403  bi_subtract(tmp3, tmp3, modulo);
404  }
405 
406  if (bi_compare_var(tmp3, modulo, FP_WORDS) >= 0) {
407  bi_subtract(tmp3, tmp3, modulo);
408  }
409 
410  bi_copy(res, tmp3);
411 }

424 void fp_exp_monty(fp_t res, const fp_t a, const bigint_t b,
425  const bigint_t modulo, const bigint_t n0, const bigint_t r2) {
426  // Montgomery exponentiation
427 
428  fp_t x_tilde;
429  fp_t a_tilde;
430  fp_t tmp;
431 
432  fp_mul_monty_var(a_tilde, a, r2, modulo, n0[0]);
433  fp_mul_monty_var(x_tilde, bi_one, r2, modulo, n0[0]);
434 
435  int num_bits = bi_get_msb(b);
436  int i;
437  for (i = num_bits; i >= 0; i--) {
438  fp_mul_monty_var(tmp, x_tilde, x_tilde, modulo, n0[0]);
439  fp_mul_monty_var(x_tilde, a_tilde, tmp, modulo, n0[0]);
440  if (bi_test_bit(b, i)) {
441  fp_copy(tmp, x_tilde);
442  } else {
443  fp_copy(x_tilde, tmp);
444  }
445  }
446  fp_mul_monty_var(res, x_tilde, bi_one, modulo, n0[0]);
447 
448  if (bi_compare(res, modulo) >= 0) {
449  bi_subtract(res, res, modulo);
450  }
451 }

641 void fp_inv_std(fp_t res, const fp_t a, const bigint_t mod) {
642 
643  // Algorithm 2.23 from Guide To ECC
644 
645  fp_t u, v;
646  fp_t x1, x2;
647  int k = 0;
648 
649  fp_copy(u, a);
650  fp_copy(v, mod);
651 
652  fp_clear(x1);
653  x1[0] = 1;
654  fp_clear(x2);
655 
656  while (bi_compare(v, bi_zero) > 0) {
657  if (bi_is_even(v)) {
658  bi_shift_right_one(v, v);
659  fp_dbl(x1, x1);
660  } else if (bi_is_even(u)) {
661  bi_shift_right_one(u, u);
662  fp_dbl(x2, x2);
663  } else if (bi_compare(v, u) >= 0) {
664  bi_subtract(v, v, u);
665  bi_shift_right_one(v, v);
666  fp_add(x2, x2, x1);
667  fp_dbl(x1, x1);
668  } else {
669  bi_subtract(u, u, v);
670  bi_shift_right_one(u, u);
671  fp_add(x1, x2, x1);
672  fp_dbl(x2, x2);
673  }
674  k++;
675  }
676 
677  if (bi_compare(x1, mod) > 0) {
678  bi_subtract(x1, x1, mod);
679  }
680 
681  // Algorithm 2.25 from Guide To ECC
682 
683 
684  if (k < BI_BITS) {
685  fp_mul(x2, x1, MONTY_PRIME_R2);
686  fp_copy(x1, x2);
687  k += BI_BITS;
688  }
689 
690  fp_mul(res, x1, MONTY_PRIME_R2);
691  if (k > BI_BITS) {
692  k = 2*BI_BITS - k;
693  bi_clear(u);
694  bi_set_bit(u, k);
695  fp_mul(x1, res, u);
696  fp_copy(res, x1);
697  }
698 }

798 void fp_rdc_2l_std(word_t *a, const bigint_t mod, const word_t *mu) {
799  // perform barrett reduction
800 
801  word_t q[2 * (FP_WORDS + 1)];
802  word_t q_p[2 * (FP_WORDS + 1)];
803  int c1, c2;
804 
805  bi_multiply_var(q, a + FP_WORDS - 1, mu, FP_WORDS + 1, FP_WORDS + 1);
806  bi_multiply_var(q_p, q + FP_WORDS + 1, mod, FP_WORDS + 1, FP_WORDS);
807  bi_subtract_var(a, a, q_p, FP_WORDS + 1);
808  bi_copy_var(q, mod, FP_WORDS);
809  q[FP_WORDS] = 0;
810 
811  // It takes at most two subtractions (this is intended to make it side-channel resistant)
812  c1 = bi_subtract_var(q_p, a, q, FP_WORDS+1);
813  c2 = bi_subtract_var(q_p+FP_WORDS+1, q_p, q, FP_WORDS+1);
814 
815 #if (ARCHITECTURE == ARCH_X86_64)
816  if (c2)
817  bi_copy(a, q_p);
818  if (!(c1 || c2))
819  bi_copy(a, q_p+FP_WORDS+1);
820 #else
821  bi_copy(a, (word_t*)(((-c1) & (int)a) | ((-c2) & (int)q_p) | (~(-c1 | -c2) & (int)(q_p+FP_WORDS+1))));
822 #endif
823 }
