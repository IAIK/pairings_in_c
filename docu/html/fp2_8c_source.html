188  // formulas:

254 {
255  // previous implementation based on Itoh Tsujii, replaced by direct formula
256  fp2_t tmp;
257  fp_t t1, t2;
258  int i;
259 
260  // calculate inverse based on inverse of subfield
261  // formula:
262  // a^{-1} = (a^r)^{-1} * a^{r-1} where r = p+1
263

268  fp_mul(t1, a[0], a[0]);

273 #else

275  fp_sub(t1, t1, t2);
276  }
277 #endif

507 void fp2_mul_qnr_std(fp2_t res, const fp2_t a) {
508  fp_t x;
509

511  // QNR: 1+i (in case p = 3 mod 8 / x0 = 7 or 11 mod 11) (ideally we use this one)
512  fp_neg(x, a[1]);
513  fp_add(res[1], a[0], a[1]); // 1*(a1*i) + i*a0
514  fp_add(res[0], a[0], x); // i*(i*a1) + a0 = a0 - a1
515 #else
516  // QNR: 1 / (1+i) (for simple impl. of bn curves (zeta!))
517  fp_neg(x, a[0]);
518  fp_add(res[0], a[0], a[1]);
519  fp_add(res[1], a[1], x);
520  fp_hlv(res[0], res[0]);
