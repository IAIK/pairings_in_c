79 void sha1_process_block(HASH_SHA1* state, uint32_t *w) {
80  int i;
81  uint32_t ST_A,ST_B,ST_C,ST_D,ST_E,ST_F,ST_K,ST_T;
82 
83  /* init state */
84  ST_A = state->H0;
85  ST_B = state->H1;
86  ST_C = state->H2;
87  ST_D = state->H3;
88  ST_E = state->H4;
89 
90  /* perform 80 rounds */
91  for (i=0; i<=79; i++) {
92  if (i<=19) {
93  ST_F = ST_D ^ (ST_B & (ST_C ^ ST_D));
94  ST_K = 0x5A827999;
95  } else if (i<=39) {
96  ST_F = ST_B ^ ST_C ^ ST_D;
97  ST_K = 0x6ED9EBA1;
98  } else if (i<=59) {
99  ST_F = (ST_B & ST_C) | (ST_D & (ST_B | ST_C));
100  ST_K = 0x8F1BBCDC;
101  } else {
102  ST_F = ST_B ^ ST_C ^ ST_D;
103  ST_K = 0xCA62C1D6;
104  }
105 
106  if (i>=16) {
107  ST_T = leftrotate(w[(i+13)&15] ^ w[(i+8)&15] ^ w[(i+2)&15] ^ w[i&15], 1);
108  w[i&15] = ST_T;
109  }
110 
111  ST_T = leftrotate(ST_A, 5) + ST_F + ST_E + ST_K + w[i&15];
112  ST_E = ST_D;
113  ST_D = ST_C;
114  ST_C = leftrotate(ST_B, 30);
115  ST_B = ST_A;
116  ST_A = ST_T;
117  }
118 
119  state->H0 += ST_A;
120  state->H1 += ST_B;
121  state->H2 += ST_C;
122  state->H3 += ST_D;
123  state->H4 += ST_E;
124 }

131 void sha1_process_message(HASH_SHA1* state, const uint8_t *message) {
132  int i;
133  uint32_t w[16];
134  uint32_t temp;
135 
136  /* copy message into the message block w */
137  for (i=0; i<16*4; i+=4) {
138  temp = (((uint32_t)message[i]) & 0xFF) << 24;
139  temp |= (((uint32_t)message[i+1]) & 0xFF) << 16;
140  temp |= (((uint32_t)message[i+2]) & 0xFF) << 8;
141  temp |= ((uint32_t)message[i+3]) & 0xFF;
142  w[i/4] = temp;
143  }
144 
145  sha1_process_block(state, w);
146 }

158 void sha1_process_message_final(HASH_SHA1* state, const uint8_t *message, uint32_t remaining_length, uint32_t total_length) {
159  int i;
160  uint32_t w[16];
161  uint32_t temp;
162 
163  /* debug_assert(total_length >= remaining_length, "total_length > remaining_length"); */
164  /* debug_assert((total_length - remaining_length) % 64 == 0, "(total_length - remaining_length) % 64 == 0"); */
165 
166  while(remaining_length >= 64) {
167  /* copy message into the message block w */
168  for (i=0; i<16*4; i+=4) {
169  temp = (((uint32_t)message[i]) & 0xFF) << 24;
170  temp |= (((uint32_t)message[i+1]) & 0xFF) << 16;
171  temp |= (((uint32_t)message[i+2]) & 0xFF) << 8;
172  temp |= ((uint32_t)message[i+3]) & 0xFF;
173  w[i/4] = temp;
174  }
175 
176  sha1_process_block(state, w);
177  message += 64;
178  remaining_length -= 64;
179  }
180 
181  for(i=0; i<16; i++)
182  w[i] = 0;
183 
184  /* copy message into the message block w */
185  for (i=0; i<remaining_length; i++) {
186  w[i/4] |= (((uint32_t)message[i]) & 0xFF) << ((~i & 3) * 8);
187  }
188 
189  /* do message padding */
190  w[remaining_length >> 2] |= 0x80L << ((~remaining_length & 3)*8);
191 
192  if(remaining_length < 56) {
193  w[15] = total_length * 8;
194  sha1_process_block(state, w);
195  } else {
196  sha1_process_block(state, w);
197 
198  for(i=0; i<15; i++)
199  w[i] = 0;
200 
201  w[15] = total_length * 8;
202  sha1_process_block(state, w);
203  }
204 }
