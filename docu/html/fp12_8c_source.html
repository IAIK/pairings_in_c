323 void fp12_mul_std(fp12_t res, const fp12_t a, const fp12_t b) {
324  fp4_t d0, d1, d2;
325  fp4_t t0;
326  fp4_t v0;
327 
328  /*
329  * Formulas
330  * D0 = a0 b0 , D1 = a1 b1 , D2 = a2 b2
331  * D0,1 = (a0 + fp4_mul(d1, a[1], b[1]); a1 )(b0 + b1 ), D0,2 = (a0 + a2 )(b0 + b2 ), D1,2 = (a1 + a2 )(b1 + b2 )
332  * C(x) = D2 x4 + (D1,2 - D1 - D2 )x3 + (D0,2 - D2 - D0 + D1 )x2 + (D0,1 - D1 - D0 )x + D0
333  *
334  */
335 
336  fp4_mul(d0, a[0], b[0]);
337  fp4_mul(d1, a[1], b[1]);
338  fp4_mul(d2, a[2], b[2]);
339 
340  fp4_add(v0, a[1], a[2]);
341  fp4_add(t0, b[1], b[2]);
342  fp4_mul(v0, (const fp2_t*)t0, (const fp2_t*)v0);
343  fp4_sub(v0, (const fp2_t*)v0, (const fp2_t*)d1);
344  fp4_sub(v0, (const fp2_t*)v0, (const fp2_t*)d2);
345  fp4_mul_adj_root(v0, (const fp2_t*)v0); // mult res[0] with qnr
346  fp4_add(v0,(const fp2_t*) d0, (const fp2_t*)v0);
347 
348  fp4_add(t0, b[0], b[1]);
349  fp4_add(res[1], a[0], a[1]);
350  fp4_mul(res[1], (const fp2_t*)res[1], (const fp2_t*)t0);
351  fp4_sub(res[1], (const fp2_t*)res[1], (const fp2_t*)d1);
352  fp4_sub(res[1], (const fp2_t*)res[1], (const fp2_t*)d0);
353  fp4_mul_adj_root(t0, (const fp2_t*)d2); // mult d2 with qnr into t0
354  fp4_add(res[1], (const fp2_t*) res[1], (const fp2_t*)t0);
355 
356  fp4_add(t0, b[0], b[2]);
357  fp4_add(res[2], a[0], a[2]);
358  fp4_mul(res[2], (const fp2_t*) res[2], (const fp2_t*) t0);
359  fp4_sub(res[2], (const fp2_t*) res[2], (const fp2_t*) d2);
360  fp4_sub(res[2], (const fp2_t*) res[2], (const fp2_t*) d0);
361  fp4_add(res[2], (const fp2_t*) res[2], (const fp2_t*) d1);
362 
363  fp4_copy(res[0], (const fp2_t*) v0);
364 }

760 void fp12_inv_std(fp12_t res, const fp12_t a) {
761  fp4_t t0, t1, t2, t3;
762 
763  // formulas: Delta = a0 - 3*a0*a1*a2*p0 + a1p0 + a2p0
764  // res0 = (a0 - a1*a2*p0) / Delta
765  // res1 = (a2*p0 - a0a1) / Delta
766  // res2 = (a1 - a0a2) / Delta
767 
768  // (a0 - a1*a2*p0)
769  fp4_sqr(t0, a[0]);
770  fp4_mul(t1, a[1], a[2]);
771  fp4_mul_adj_root(t2, (const fp2_t*) t1);
772  fp4_sub(t1, (const fp2_t*) t0, (const fp2_t*) t2);
773 
774  // (a2*p0 - a0a1)
775  fp4_sqr(t0, a[2]);
776  fp4_mul_adj_root(t3, (const fp2_t*) t0);
777  fp4_mul(t2, a[0], a[1]);
778  fp4_sub(t2, (const fp2_t*) t3, (const fp2_t*) t2);
779 
780  // (a1 - a0a2)
781  fp4_sqr(t0, a[1]);
782  fp4_mul(t3, a[0], a[2]);
783  fp4_sub(t3, (const fp2_t*) t0, (const fp2_t*) t3);
784 
785  // res1 = a1 * (a1 - a0a2) * p0
786  fp4_mul(t0, a[1], (const fp2_t*) t3);
787  fp4_mul_adj_root(res[1], (const fp2_t*) t0);
788 
789  // res0 = a0 * (a0 - a1*a2*p0)
790  fp4_mul(res[0], a[0], (const fp2_t*) t1);
791 
792  // res2 = a2 * (a2*p0 - a0a1) * p0
793  fp4_mul(t0, a[2], (const fp2_t*) t2);
794  fp4_mul_adj_root(res[2],(const fp2_t*) t0);
795 
796  // t0 = (a0 * (a0 - a1*a2*p0) + a1 * (a1 - a0a2) * p0 + a2 * (a2*p0 - a0a1) * p0)^{-1}
797  // = (a0 - a0a1a2p0 + a1p0 - a0a1a1p0 + a2p0 - a0a1a2p0)^{-1}
798  // = (a0 - 3*a0*a1*a2*p0 + a1p0 + a2p0)^{-1} = Delta^{-1}
799  fp4_add(t0, (const fp2_t*) res[0], (const fp2_t*) res[1]);
800  fp4_add(t0, (const fp2_t*) t0, (const fp2_t*) res[2]);
801  fp4_inv(t0, (const fp2_t*) t0);
802 
803  fp4_mul(res[0], (const fp2_t*) t1, (const fp2_t*) t0);
804  fp4_mul(res[1], (const fp2_t*) t2, (const fp2_t*) t0);
805  fp4_mul(res[2], (const fp2_t*) t3, (const fp2_t*) t0);
806 }
