53 void sgs_init(length_t number_of_members, group_public_key *gpk, group_master_secret_key *gmsk, group_secret_key *gsk)

67

77  fp_inv_n(inv_2, gmsk->xi2);

81  ecfp_mul(&gpk->v, &gpk->h, inv_2);

86

292 sbyte sgs_verify(group_public_key gpk, sdh_signiture sig, const char *message) {
293  bigint_t bi_tmp, bi_tmp1, c;
294  ecpoint_fp G1_tmp1, G1_tmp2, R_1, R_2, R_4, R_5;
295  fp12_t GT_tmp1, GT_tmp2, GT_tmp3, R_3;

301  // R1
302  bi_subtract(bi_tmp, EC_PARAM_N, sig.c);
303  ecfp_mul(&G1_tmp1, &sig.T_1, bi_tmp);

338 
339  fp12_mul_std(R_3, GT_tmp3, GT_tmp2);
340 
341  // R4
342  bi_subtract(bi_tmp, EC_PARAM_N, sig.s_delta1);
343  ecfp_mul(&G1_tmp1, &gpk.u, bi_tmp);
