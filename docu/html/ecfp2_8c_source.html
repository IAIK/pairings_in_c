118 void ecfp2_add_proj_std(ecfp2_proj_pt res, const ecfp2_proj_pt a, const ecfp2_pt b) {
119  // calculation based on jacobian coordinates
120  // b in projective coordinates is (b.x, b.y, 1)
121 
122  if (a->infinity && !b->infinity) { // P + Infinity = P
123  fp2_copy(res->x, (const fp_t*) b->x);
124  fp2_copy(res->y, (const fp_t*) b->y);
125  fp_copy(res->z[0], FP_ONE);
126  fp_copy(res->z[1], bi_zero);
127  res->infinity = 0;
128  return;
129  } else if (!a->infinity && b->infinity) {
130  fp2_copy(res->x, (const fp_t*) a->x);
131  fp2_copy(res->y, (const fp_t*) a->y);
132  fp_copy(res->z[0], FP_ONE);
133  fp_copy(res->z[1], bi_zero);
134  res->infinity = 0;
135  return;
136  } else if (a->infinity && b->infinity) {
137  res->infinity = 1;
138  return;
139  }
140 
141  fp2_t t0, t1, t2, t3;
142 
143  fp2_sqr(t0, (const fp_t*) a->z); // t0 = z²
144  fp2_mul(t1, (const fp_t*) t0, (const fp_t*) a->z); // t1 = z³
145 
146  fp2_mul(t2, (const fp_t*) t0, (const fp_t*) b->x); // t2 = b->x*z²
147  fp2_sub(t0, (const fp_t*) t2, (const fp_t*) a->x); // t0 = b->x*z² - a->x
148 
149  fp2_mul(t1, (const fp_t*) t1, (const fp_t*) b->y);
150  fp2_sub(t1, (const fp_t*) t1, (const fp_t*) a->y); // t1 = b->y*z³ - a->y
151 
152  if (bi_compare(t0[0], bi_zero) == 0 && bi_compare(t0[1], bi_zero) == 0) {
153  if (bi_compare(t1[1], bi_zero) == 0 && bi_compare(t1[1], bi_zero) == 0) {
154  // point a is equal to point b (doubling!)
155  ecfp2_dbl_proj(res, a);
156  } else { // otherwise its point (1,1,0) -> infinity
157  res->infinity = 1;
158  }
159  return;
160  } else {
161  res->infinity = 0;
162  }
163 
164  fp2_mul(res->z, (const fp_t*) t0, (const fp_t*) a->z); // res->z = (b->x*z² - a->x) * a->z
165 
166  fp2_add(t2, (const fp_t*) t2, (const fp_t*) a->x); // t2 = b->x * z² + a->x
167  fp2_sqr(t3, (const fp_t*) t0); // t3 = (b->x*z² - a->x)²
168  fp2_mul(t2, (const fp_t*) t2, (const fp_t*) t3); // t2 = (b->x*z² - a->x)² * (b->x * z² + a->x)
169  fp2_mul(t0, (const fp_t*) t0, (const fp_t*) t3); // t0 = (b->x*z² - a->x)³
170  fp2_mul(t0, (const fp_t*) t0, (const fp_t*) a->y); // t0 = a->y * (b->x*z² - a->x)³
171  fp2_mul(t3, (const fp_t*) t3, (const fp_t*) a->x); // t3 = a->x * (b->x*z² - a->x)²
172 
173  fp2_sqr(res->x, (const fp_t*) t1); // res->x = (b->y*z³ - a->y)²
174  fp2_sub(res->x, (const fp_t*) res->x, (const fp_t*) t2);// res->x = (b->y*z³ - a->y)² -
175  // (b->x*z² - a->x)² * (b->x * z² + a->x)
176  fp2_sub(t3, (const fp_t*) t3, (const fp_t*) res->x); // a->x * (b->x*z² - a->x)² - res->x
177  fp2_mul(t3, (const fp_t*) t3, (const fp_t*) t1);
178  fp2_sub(res->y, (const fp_t*) t3, (const fp_t*) t0); // (b->y*z³ - a->y)(a->x * (b->x*z² - a->x)² - res->x) -
179  // a->y * (b->x*z² - a->x)³
180 }

248  // calculation based on jacobian coordinates
249 
250  if (a->infinity) {
251  res->infinity = 1;
252  return;
253  } else {
254  res->infinity = 0;
255  }
256 
257  fp2_t t0, t1, t2;
258 
259  fp2_sqr(t0, (const fp_t*) a->x);
260 
261 #ifdef OPTIMIZE_ZERO_A
262  fp2_dbl(t1, (const fp_t*) t0);
263  fp2_add(t1, (const fp_t*) t1, (const fp_t*) t0); // t1 = a*z⁴ + 3x²
264 #else
265  fp2_sqr(t1, a->z);
266  fp2_sqr(t1, (const fp_t*) t1); // t1 = z⁴
267 
268  fp_clear(t2[1]); // calc a' of twisted curve
269  fp_copy(t2[0], EC_PARAM_A);
270  fp2_div_qnr(t2, (const fp_t*) t2);
271  fp2_mul(t1, t1, t2); // t1 = a*z⁴
272 
273  fp2_add(t1, t1, t0);
274  fp2_add(t1, t1, t0);
275  fp2_add(t1, t1, t0); // t1 = a*z⁴ + 3x²
276 #endif
277 
278  fp2_dbl(t0, (const fp_t*) a->y); // t0 = 2 * y
279  fp2_mul(res->z, (const fp_t*) t0, (const fp_t*) a->z); // res->z = 2*y*z
280 
281  fp2_sqr(t0, (const fp_t*) t0); // t0 = 4 * y²
282  fp2_mul(t2, (const fp_t*) t0, (const fp_t*) a->x); // t2 = 4*x*y²
283 
284  fp2_copy(res->y, (const fp_t*) t1); // res->y = (3x² + az⁴)
285  fp2_dbl(t1, (const fp_t*) t2); // t1 = 8*x*y²
286

289

291  fp2_mul(res->y, (const fp_t*) res->y, (const fp_t*) t2);// res->y = (3x² + az⁴)(4xy² - res->x)
292 
293  fp2_sqr(t0, (const fp_t*) t0);
294  fp2_hlv(t0, (const fp_t*) t0); // t0 = 8 * y⁴
295 
296  fp2_sub(res->y, (const fp_t*) res->y, (const fp_t*) t0);// res->y = (3x² + az⁴)(4xy² - res->x) - 8y⁴
297 }

561 void ecfp2_mul_montyladder_std(ecfp2_pt res, const ecfp2_pt a, const fp_t k) {
562  int msb = bi_get_msb(k);
563  int i;
564  fp2_t x1, x2, z;

566  word_t bit;
567 
568  x[0] = x1;
569  x[1] = x2;
570

573

594  ecfp2_add_dbl_coz_std(x[~bit & 1], x[bit], z, (const fp_t*) a->x);
595  }
596

598 
599  if (ecfp2_verify_homogeneous_projective_coordinates(x[0], x[1], z))
600  return;
601 
602  fp2_inv(z, (const fp_t*) z);
603  fp2_mul(res->x, (const fp_t*) x[0], (const fp_t*) z);
604  fp2_mul(res->y, (const fp_t*) x[1], (const fp_t*) z);
605  res->infinity = 0;

608  return;

812 void ecfp2_hash_to_point_std(ecfp2_pt res, const bigint_t t) {
813  fp_t sqrtM3;
814  fp2_t w, t0;
815  fp2_t x[3];
816  ecpoint_fp2 px;
817  ecpoint_fp2_proj tp;
818 
819  // Firstly: Hash to a random point
820 
821  fp_dbl(sqrtM3, FP_ONE);
822  fp_add(sqrtM3, sqrtM3, FP_ONE);
823  fp_neg(sqrtM3, sqrtM3);
824  fp_sqrt(sqrtM3, sqrtM3); // sqrt(-3)
825 
826  fp_clear(w[1]);
827  fp_sqr(w[0], t);
828  fp_add(w[0], w[0], FP_ONE);
829 
830  fp_clear(t0[1]); // calc b' of twisted curve
831  fp_copy(t0[0], EC_PARAM_B);
832  fp2_div_qnr(t0, (const fp_t*) t0);
833 
834  fp2_add(w, (const fp_t*) w, (const fp_t*) t0);
835  fp2_inv(w, (const fp_t*) w);
836  fp2_mulfp(w, (const fp_t*) w, sqrtM3);
837  fp2_mulfp(w, (const fp_t*) w, (const word_t*) t);
838 
839  fp2_sqr(x[2], (const fp_t*) w);
840  fp2_inv(x[2], (const fp_t*) x[2]);
841  fp_add(x[2][0], x[2][0], FP_ONE);
842 
843  fp_clear(x[0][1]);
844  fp_sub(x[0][0], sqrtM3, FP_ONE);
845  fp2_hlv(x[0], (const fp_t*) x[0]);
846  fp2_mulfp(w, (const fp_t*) w, (const word_t*) t);
847  fp2_sub(x[0], (const fp_t*) x[0], (const fp_t*) w);
848 
849  fp_clear(x[1][1]);
850  fp_neg(x[1][0], FP_ONE);
851  fp2_sub(x[1], (const fp_t*) x[1], (const fp_t*) x[0]);
852 
853  // implementation without blinding (as suggested by Fouque, Tibouchi)
854  fp2_sqr(w, (const fp_t*) x[0]);
855  fp2_mul(w, (const fp_t*) w, (const fp_t*) x[0]);
856  fp2_add(w, (const fp_t*) w, (const fp_t*) t0); // t0 = b'
857 
858  int alpha = fp2_legendre((const fp_t*) w);
859 
860  fp2_sqr(w, (const fp_t*) x[1]);
861  fp2_mul(w, (const fp_t*) w, (const fp_t*) x[1]);
862  fp2_add(w, (const fp_t*) w, (const fp_t*) t0);
863 
864  int beta = fp2_legendre((const fp_t*) w);
865 
866  int idx = ((alpha-1)*beta % 3);
867  if (idx < 0)
868  idx+=3;
869 
870  res->infinity = 0;
871  fp2_copy(res->x, (const fp_t*) x[idx]);
872  fp2_sqr(w, (const fp_t*) x[idx]);
873  fp2_mul(w, (const fp_t*) w, (const fp_t*) x[idx]);
874  fp2_add(w, (const fp_t*) w, (const fp_t*) t0);
875  fp2_sqrt(res->y, (const fp_t*) w);
876 
877  if (fp_legendre(t) == -1)
878  fp2_neg(res->y, (const fp_t*) res->y);
879 
880  // Secondly: multiply by cofactor to have correct order
881 
882  ecfp2_mul(&px, res, EC_PARAM_X);
883 #ifdef NEGATIVE_PARAM_X
884  ecfp2_neg_affine(&px);
885 #endif
886  ecfp2_frobenius_map(res, res, 3);
887  ecfp2_get_jacobian_projective(&tp, res);
888  ecfp2_add_proj(&tp, &tp, &px);
889 
890  ecfp2_frobenius_map(res, &px, 2);
891  ecfp2_add_proj(&tp, &tp, res);
892 
893  ecfp2_dbl_affine(res, &px);
894  ecfp2_add_affine(res, res, &px);
895  ecfp2_frobenius_map(res, res, 1);
896  ecfp2_add_proj(&tp, &tp, res);
897 
898  ecfp2_get_affine_from_jacobian(res, &tp);
899 }
