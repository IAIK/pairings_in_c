55 sbyte hwang_sign(hwang_signature_ptr sig, hwang_public_parameters_ptr data, hwang_signing_key_ptr usk)
56 {
57  hashState state;
58  unsigned char block[64];
59  unsigned char hash[HASH_BYTES];
60  unsigned char msg[1] = {0x01};
61 
62  bigint_t alpha, r_alpha, r_gamma, r_x, r_y;
63  word_t bigint_tmp_dbl[2*BI_WORDS], bigint_tmp_dbl2[2*BI_WORDS];
64 
65  bigint_t gamma, bi_tmp;
66 
67  ecpoint_fp G1_tmp1, G1_tmp2, R_1, R_3;
68 
69  fp12_t R_2, GT_tmp1;
70 
71  bi_clear_var_std(bigint_tmp_dbl, 2*BI_WORDS); bi_clear_var_std(bigint_tmp_dbl2, 2*BI_WORDS);
72 
73  // Generate blinding values.
74  do {
75  cprng_get_bytes(alpha, BI_BYTES); fp_rdc_n(alpha);
76  } while (bi_compare(alpha, bi_zero) == 0);
77 
78  do {
79  cprng_get_bytes(r_alpha, BI_BYTES); fp_rdc_n(r_alpha);
80  } while (bi_compare(r_alpha, bi_zero) == 0);
81 
82  do {
83  cprng_get_bytes(r_gamma, BI_BYTES); fp_rdc_n(r_gamma);
84  } while (bi_compare(r_gamma, bi_zero) == 0);
85 
86  do {
87  cprng_get_bytes(r_x, BI_BYTES); fp_rdc_n(r_x);
88  } while (bi_compare(r_x, bi_zero) == 0);
89 
90  do {
91  cprng_get_bytes(r_y, BI_BYTES); fp_rdc_n(r_y);
92  } while (bi_compare(r_y, bi_zero) == 0);
93 
94 #if DEBUG_PRINT == 1
95  PRINT_BIGINT("alpha: ", alpha);
96  PRINT_BIGINT("order: ", EC_PARAM_N);
97  PRINT_BIGINT("r_alpha: ", r_alpha);
98  PRINT_BIGINT("r_gamma: ", r_gamma);
99  PRINT_BIGINT("r_x: ", r_x);
100  PRINT_BIGINT("r_y: ", r_y);
101 #endif
102 
103  bi_copy_var(bigint_tmp_dbl2, usk->z, BI_WORDS);
104  bi_multiply(bigint_tmp_dbl, usk->x, alpha);
105  bi_subtract_dbl_len(bigint_tmp_dbl, bigint_tmp_dbl, bigint_tmp_dbl2);
106  fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
107  bi_copy_var_std(gamma, bigint_tmp_dbl, BI_WORDS);
108 
109  ecfp_mul(&sig->D_1, &data->u, alpha); // D1 = u^{alpha}
110 
111  ecfp_mul(&sig->D_2, &data->w, alpha);
112  ecfp_add_affine(&sig->D_2, &sig->D_2, &usk->A); // D2 = A * w^{alpha}
113 
114  ecfp_mul(&sig->D_3, &data->g, usk->y); // D3 = g^y
115  ecfp_mul(&G1_tmp1, &data->d, alpha); // tmp = d^{alpha}
116  ecfp_add_affine(&sig->D_3, &sig->D_3, &G1_tmp1); // D3 = g^y * d^{alpha]}
117 
118  ecfp_mul(&R_1, &data->u, r_alpha); // R1 = u^{r_alpha}
119 
120 #if DEBUG_PRINT == 1
121  PRINT_G1("[sign] R_1: ", R_1);
122 #endif
123 
124  // R_2 = e(D_2^{r_x} * w^{-r_{gamma}} * g_2^{r_y}, h_1) * e(w^{-r_{alpha}}, h_{theta});
125  bi_subtract(bi_tmp, data->order_p, r_gamma);
126 
127  ecfp_mul(&G1_tmp1, &sig->D_2, r_x);
128  ecfp_mul(&G1_tmp2, &data->w, bi_tmp);
129  ecfp_add_affine(&G1_tmp1, &G1_tmp1, &G1_tmp2);
130 
131  ecfp_mul(&G1_tmp2, &data->g2, r_y);
132  ecfp_add_affine(&G1_tmp1, &G1_tmp1, &G1_tmp2);
133 
134  pbc_map_opt_ate(R_2, &G1_tmp1, &data->h1); // e(D_2^{r_x} * w^{-r_{gamma}} * g_2^{r_y}, h_1)
135 
136  bi_subtract(bi_tmp, data->order_p, r_alpha);
137  ecfp_mul(&G1_tmp1, &data->w, bi_tmp);
138 
139  pbc_map_opt_ate(GT_tmp1, &G1_tmp1, &data->h_theta);
140 
141  fp12_mul(R_2, R_2, GT_tmp1);
142 
143 #if DEBUG_PRINT == 1
144  PRINT_GT("[sign] R_2: ", R_2);
145 #endif
146 
147  //R_3 = g^{r_y} * d^{r_{alpha}}
148  ecfp_mul(&R_3, &data->g, r_y);
149  ecfp_mul(&G1_tmp1, &data->d, r_alpha);
150  ecfp_add_affine(&R_3, &R_3, &G1_tmp1);
151 
152 #if DEBUG_PRINT == 1
153  PRINT_G1("[sign] R_3: ", R_3);
154 #endif
155 
156  hash_init(&state);
157 
158  // hash message M
159  memset(block, 0, 64); memcpy(block, msg, 1); hash_update(&state, block, 64);
160 
161  // hash D1
162  hash_update_G1(&state, &sig->D_1);
163 
164  // hash D2
165  hash_update_G1(&state, &sig->D_2);
166 
167  // hash D3
168  hash_update_G1(&state, &sig->D_3);
169 
170  // hash R1
171  hash_update_G1(&state, &R_1);
172 
173  // hash R2
174  hash_update_GT(&state, R_2);
175 
176  // hash R3
177  hash_update_G1(&state, &R_3);
178 
179  hash_final(&state , hash, HASH_BYTES);
180 
181  bi_clear_var_std(bigint_tmp_dbl, 2*BI_WORDS); bi_copy_var_std(bigint_tmp_dbl, (const word_t *) hash, HASH_BYTES);
182  fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
183  bi_copy_var_std(sig->c, bigint_tmp_dbl, BI_WORDS);
184 
185 
186  bi_multiply(bigint_tmp_dbl, sig->c, alpha); fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
187  bi_copy_var_std(sig->s_alpha, bigint_tmp_dbl, BI_WORDS);
188  bi_add(sig->s_alpha, sig->s_alpha, r_alpha); fp_rdc_n(sig->s_alpha);
189 
190  bi_multiply(bigint_tmp_dbl, sig->c, usk->x); fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
191  bi_copy_var_std(sig->s_x, bigint_tmp_dbl, BI_WORDS);
192  bi_add(sig->s_x, sig->s_x, r_x); fp_rdc_n(sig->s_x);
193 
194  bi_multiply(bigint_tmp_dbl, sig->c, gamma); fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
195  bi_copy_var_std(sig->s_gamma, bigint_tmp_dbl, BI_WORDS);
196  bi_add(sig->s_gamma, sig->s_gamma, r_gamma); fp_rdc_n(sig->s_gamma);
197  bi_multiply(bigint_tmp_dbl, sig->c, usk->y); fp_rdc_2l_var(bigint_tmp_dbl, data->order_p, EC_PARAM_MU_N);
198  bi_copy_var_std(sig->s_y, bigint_tmp_dbl, BI_WORDS);
199  bi_add(sig->s_y, sig->s_y, r_y); fp_rdc_n(sig->s_y);
200 
201 #if DEBUG_PRINT == 1
202  print(" --------- Signature --------- \n");
203  PRINT_BIGINT("c: ", sig->c);
204  PRINT_BIGINT("s_alpha: ", sig->s_alpha);
205  PRINT_BIGINT("s_x :", sig->s_x);
206  PRINT_BIGINT("s_gamma: ", sig->s_gamma);
207  PRINT_BIGINT("s_y: ", sig->s_y);
208 
209  PRINT_G1("D_1: ", sig->D_1);
210  PRINT_G1("D_2: ", sig->D_2);
211  PRINT_G1("D_3: ", sig->D_3);
212 #endif
213 
214  return 0;
215 }
