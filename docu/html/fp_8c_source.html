394 void fp_mul_monty_std(fp_t res, const fp_t a, const fp_t b, const bigint_t modulo, const bigint_t n0) {
395 
396  // Montgomery multiplication
397  word_t tmp[2 * FP_WORDS];
398  word_t tmp2[3 * FP_WORDS];
399  word_t tmp3[2 * FP_WORDS];
400 
401  // Multiplication
402  bi_multiply(tmp, a, b); // 2*FP_WORDS
403 
404  // Reduction
405  // t * n0
406  bi_multiply_var(tmp2, tmp, n0, 2 * FP_WORDS, FP_WORDS); // 3*FP_WORDS
407 
408  // Multiply by n(=modulo)
409  bi_multiply(tmp3, tmp2, modulo); // 2*FP_WORDS
410 
411  // Add t
412  int carry = bi_add_var(tmp, tmp, tmp3, 2 * FP_WORDS); // 2*FP_WORDS
413 
414  // Divide by R(right-shift by R bit)
415  bi_shift_right_var(tmp3, tmp, FP_BITS,
416  2 * FP_WORDS); // FP_WORDS
417 
418  if (carry) {
419  bi_subtract(tmp3, tmp3, modulo);
420  }
421 
422  if (bi_compare_var(tmp3, modulo, FP_WORDS) >= 0) {
423  bi_subtract(tmp3, tmp3, modulo);
424  }
425 
426  bi_copy(res, tmp3);
427 }

575 void fp_exp_slide_var(fp_t res, const fp_t a, const bigint_t b, const bigint_t mod, const bigint_t mu) {
576  fp_t tab[FP_EXP_SLW_TBL_SZ]; // precomputation table
577  int i;
578 
579  bi_copy(tab[0], a);
580  fp_sqr_var(res, a, mod, mu);
581 
582  // precomputation of table: a, a^3, a^5,...
583  for (i = 1; i < FP_EXP_SLW_TBL_SZ; i++) {
584  fp_mul_var(tab[i], tab[i - 1], res, mod, mu);
585  }
586 
587  //fp_set_dig(t, 1);
588  bi_clear(res); res[0] = 1;
589 
590  int exp;
591  int slw_len;
592  int bit;
593  int step;
594 
595  // "step" represents the number of bits that was looked at
596  for (bit = bi_get_msb(b); bit >= 0; bit-=step) {
597  if (bi_test_bit(b, bit)) {
598 
599  slw_len = 1; // length of sliding window found
600  exp = 1; // index of the exponent in the sliding window
601  i = 1; // running idx to find sliding window
602 
603  // analyze exponent
604  while (i < FP_EXP_SLW_WIDTH && (bit-i) >= 0) {
605  if (bi_test_bit(b, bit-i)) {
606  exp = (exp << (i - slw_len + 1)) + 1;
607  slw_len = i+1;
608  }
609  i++;
610  }
611  step = i;
612 
613  // pre-processing: square
614  for (i = 0; i < slw_len; i++) {
615  fp_sqr_var(res, res, mod, mu);
616  }
617 
618  // multiply precomupted
619  fp_mul_var(res, res, tab[exp >> 1], mod, mu);
620 
621  // post-processing: square
622  for (i = slw_len; i < step; i++) {
623  fp_sqr_var(res, res, mod, mu);
624  }
625  } else {
626  // simple square in case of zero
627  fp_sqr_var(res, res, mod, mu);
628  step = 1;
629  }
630  }
631 }

812  // Algorithm 2.23 from Guide To ECC
813 
814  fp_t u, v;
815  fp_t x1, x2;
816  int k = 0;
817 
818  fp_copy(u, a);
819  fp_copy(v, mod);
820 
821  fp_clear(x1);
822  x1[0] = 1;
823  fp_clear(x2);
824 
825  while (bi_compare(v, bi_zero) > 0) {
826  if (bi_is_even(v)) {
827  bi_shift_right_one(v, v);
828  fp_dbl(x1, x1);
829  } else if (bi_is_even(u)) {

988  bi_copy(a, q_p+FP_WORDS+1);
