67  // this multiplication needs to have constant runtime
68  fp_mul_barett_var(factor, id_bi, s, EC_PARAM_N, EC_PARAM_MU_N);

70  // s*g3 + factor*g1 ... => interleaved calculation?
71  ecfp_mul(&(cipher->c1), (const ecfp_pt) &(BB1_PUBLIC.g3), s);

74  ecfp_add_affine(&(cipher->c1), &(cipher->c1), &(cipher->c0));
