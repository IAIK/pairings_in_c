45 int bi_get_nafw_std(sbyte *nafw, word_t *a, word_t width) {
46  uint i = 0;
47  uint mod = (1 << width);
48  uint mod_mask = (1 << width) - 1;
49  uint bit_mask = (1 << (width-1));
50  word_t carry, tmp;
51 
52  carry = 0;
53 
54  // it is simplified as width is expected not to be bigger than the word size
55  while (bi_compare(a, bi_zero) > 0) {
56  if (bi_is_odd(a)) {
57  nafw[i] = a[0] & mod_mask;
58  if (nafw[i] & bit_mask)
59  nafw[i] = nafw[i] - mod;
60  tmp = a[0] >> (BITS_PER_WORD-1);
61  a[0] -= (word_t)nafw[i];
62  carry = tmp & ~(a[0] >> (BITS_PER_WORD-1));
63  } else {
64  nafw[i] = 0;
65  }
66 
67  // carry propagation
68  tmp = a[1] & 0x01 & carry;
69  a[1] = a[1] ^ carry;
70  carry = tmp;
71 
72  bi_shift_right_one(a, a);
73  i++;
74  }
75 
76  return i;
77 }

295 void bi_shift_right_var_std(word_t *res, const word_t *a, const uint right, const length_t length) {
296  int i;
297  uint wordNum = right >> LD_BITS_PER_WORD;
298  uint bitNum = right & (BITS_PER_WORD-1);
299  word_t shift_in;
300 
301  for(i = length-1; i >= (int)(length-wordNum); i--)
302  res[i] = 0;
303 
304  if (bitNum == 0) {
305  a += wordNum;
306  for (i = 0; i < (int)(length-wordNum); i++) {
307  *(res+i) = *(a+i);
308  }
309  } else {
310  a += wordNum;

312  shift_in = *(a+i+1) << (BITS_PER_WORD - bitNum);
313  *(res+i) = (*(a+i) >> bitNum) | shift_in;
314  }

442 void bi_multiply_var_std(word_t *result, const word_t *a, const word_t *b, const length_t length_a, const length_t length_b)
443 {
444  dword_t tmp;
445  word_t carry;
446  uint i, j;
447  bi_clear_var(result, length_a + length_b);
448 
449  // many carries ... bad performance -> comba is better :)
