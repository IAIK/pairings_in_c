167 void KeccakF( tKeccakLane * state, const tKeccakLane *in, int laneCount )
168 {

170  tKeccakLane temp;
171  tKeccakLane BC[5];
172 
173  #if (PLATFORM_BYTE_ORDER == IS_LITTLE_ENDIAN) || (cKeccakB == 200)
174  while ( --laneCount >= 0 )
175  {
176  state[laneCount] ^= in[laneCount];
177  }
178  #else
179  temp = 0; /* please compiler */
180  while ( --laneCount >= 0 )
181  {
182  for ( x = 0; x < sizeof(tKeccakLane); ++x )
183  {
184  temp <<= 8;
185  temp |= ((char*)&in[laneCount])[x];
186  }
187  state[laneCount] = temp;
188  }
189  #endif
190 
191  #define round laneCount
192  for( round = 0; round < cKeccakNumberOfRounds; ++round )
193  {
194  // Theta
195  for ( x = 0; x < 5; ++x )
196  {
197  BC[x] = state[x] ^ state[5 + x] ^ state[10 + x] ^ state[15 + x] ^ state[20 + x];
198  }
199  for ( x = 0; x < 5; ++x )
200  {
201  temp = BC[KeccakF_Mod5[x+4]] ^ ROL(BC[KeccakF_Mod5[x+1]], 1);
202  for ( y = 0; y < 25; y += 5 )
203  {
204  state[y + x] ^= temp;
205  }
206  }
207 
208  // Rho Pi
209  temp = state[1];

213  state[KeccakF_PiLane[x]] = ROL( temp, KeccakF_RotationConstants[x] );

215  }
216 
217  // Chi
218  for ( y = 0; y < 25; y += 5 )
219  {
220  BC[0] = state[y + 0];
221  BC[1] = state[y + 1];
222  BC[2] = state[y + 2];

249 {

251  if ( (state->bitsInQueue < 0) || ((state->bitsInQueue % 8) != 0) )
252  {

254  return ( FAIL );

272  }
