564 void pbc_map_opt_ate_optimized_miller_std(fp12_t res, ecfp_pt p, ecfp2_pt q) {

566  ecpoint_fp2 q1;
567  int len, i;
568  fp12_t a;
569 
570  // Actually, the pairing over BN curves is defined as
571  // E(F_p) x E'(F_p12) -> F_p12
572  // but there is an efficient isomorphism from E(F_p2) to E(F_p12)
573 
574  ecfp2_get_homogeneous_projective(&t, q);
575  fp12_clear(res); res[0][0][0][0] = 1;
576  fp12_clear(a);
577 
578 #ifdef OPT_ATE_LOOP_CONSTANT
579  const word_t *s = OPTATE_LOOP_CONST;
580 #else

585  #else

587  #endif
588 #endif

595  fp12_sqr(res, (const fp4_t*) res);
596  pbc_point_dbl_line_costello_eval(a, &t, &t, p);
597  fp12_mul_sparse(res, (const fp4_t*) res, (const fp2_t*) a[0], (const fp_t*) a[1][0]);
598
