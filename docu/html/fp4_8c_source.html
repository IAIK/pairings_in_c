112 void fp4_mul_std(fp4_t res, const fp4_t a, const fp4_t b) {
113  fp2_t x, y;
114  // Multiplication via Karatsuba method
115  // formulas:
116  // res0 = a0*b0 + zeta * a1*b1
117  // res1 = (a1+a0)(b1+b0) - a0*b0 - a1*b1
118  // zeta: a quadratic non-residue in F_{p^2} used to construct F_{p^4} towering F_{p^2}
119  // by defining F_{p^2}[Y]/(Y^2 + zeta)
120 
121  fp2_add(x, a[1], a[0]); // x = a1 + a0
122  fp2_add(y, b[1], b[0]); // y = b1 + b0
123  fp2_mul(y, (const fp_t*) x, (const fp_t*) y); // y = x * y = (a1+a0)(b1+b0)
124 
125  fp2_mul(res[0], a[0], b[0]); // res0 = a0*b0
126  fp2_mul(x, a[1], b[1]); // x = a1*b1
127 
128  fp2_sub(res[1], (const fp_t*) y, (const fp_t*) res[0]); // res1 = y - res0 = (a1+a0)(b1+b0) - a0*b0
129  fp2_sub(res[1], (const fp_t*) res[1], (const fp_t*) x); // res1 = res1 - x = (a1+a0)(b1+b0) - a0*b0 - a1*b1
130 
131  fp2_mul_qnr(x, (const fp_t*) x);
132  fp2_add(res[0], (const fp_t*) res[0], (const fp_t*) x);
133 }

152 void fp4_exp_std(fp4_t res, const fp4_t a, const bigint_t b) {
153  fp4_t tmp;
154  int i;
155 
156  fp2_clear(res[1]);
157  fp_clear(res[0][1]);
158  fp_copy(res[0][0], FP_ONE);
159 
160  for (i = bi_get_msb(b); i >= 0; i--) {
161  fp4_sqr(res, (const fp2_t*) res);
162  if (bi_test_bit(b, i)) {
163  fp4_mul(res, (const fp2_t*) res, (const fp2_t*) a);
164  } else { // else side channel dummy
165  fp4_mul(tmp, (const fp2_t*) res, (const fp2_t*) a);
166  }
167  }

175 void fp4_inv_std(fp4_t res, const fp4_t a) {
176  fp2_t t0, t1;
177 
178  // formulas: res0 = a0 / (a0^2 - p0*a1^2)
179  // res1 = -a1 / (a0^2 - p0*a1^2)
180  // for: P(x) = x² - p0
181 
182  fp2_sqr(t0, a[0]);
183  fp2_sqr(t1, a[1]);
184  fp2_mul_qnr(t1, (const fp_t*) t1);
185  fp2_sub(t0, (const fp_t*) t0, (const fp_t*) t1);
186  fp2_inv(t0, (const fp_t*) t0);
187 
188  fp2_mul(res[0], a[0], (const fp_t*) t0);
189  fp2_mul(res[1], a[1], (const fp_t*) t0);
190  fp2_neg(res[1], (const fp_t*) res[1]);
191 }

201  // map the coefficients

203  fp2_frobenius_map(res[1], a[1], 1);

224 void fp4_frobenius_map_std(fp4_t res, const fp4_t a, const word_t i) {
225  fp2_t q, q_exp;
226  bigint_t e;
227 
228  fp2_clear(q);
229  q[0][0] = 1;
230  fp2_mul_qnr(q, (const fp_t*) q);
231  bi_copy(e, PRIME_P);
232  bi_subtract(e, e, bi_one);
233  bi_shift_right_one(e, e);
234  fp2_exp(q_exp, (const fp_t*) q, (const word_t*) e); // qnr^{(p-1)/2} (maybe frobenius, inverse + root is cheaper?)
235 
236  // map the coefficients
237  fp2_frobenius_map(res[0], a[0], i);
238  fp2_frobenius_map(res[1], a[1], 1);
239 
240  // multiply with exponentiated qnr
241  fp2_mul(res[1], (const fp_t*) res[1], (const fp_t*) q_exp);
242 
243  // iterate the map (map for x^{p^i})
244  word_t j = i;
245  while (j > 1) {
246  fp2_frobenius_map(res[1], (const fp_t*) res[1], 1);
247  fp2_mul(res[1], (const fp_t*) res[1], (const fp_t*) q_exp);
248  j--;
